Synopsis
--------
- This module provides declarative management of VLANs on Arista EOS network devices.

Examples
--------

.. code-block:: yaml

    - name: Create vlan
      arista.eos.eos_vlan:
        vlan_id: 4000
        name: vlan-4000
        state: present

    - name: Add interfaces to vlan
      arista.eos.eos_vlan:
        vlan_id: 4000
        state: present
        interfaces:
        - Ethernet1
        - Ethernet2

    - name: Check if interfaces is assigned to vlan
      arista.eos.eos_vlan:
        vlan_id: 4000
        associated_interfaces:
        - Ethernet1
        - Ethernet2

    - name: Suspend vlan
      arista.eos.eos_vlan:
        vlan_id: 4000
        state: suspend

    - name: Unsuspend vlan
      arista.eos.eos_vlan:
        vlan_id: 4000
        state: active

    - name: Create aggregate of vlans
      arista.eos.eos_vlan:
        aggregate:
        - vlan_id: 4000
        - {vlan_id: 4001, name: vlan-4001}
